• 配置gradle环境


#1. 在"系统属性对话框里"要先设置环境变量和路径:
set M2_HOME=e:\WJW_APP\PStart\Java\maven3
set GRADLE_USER_HOME=z:\GRADLE_USER_HOME
set path=%path%;C:\ProgramData\Oracle\Java\javapath;c:\Java\jdk1.8\bin;e:\WJW_APP\PStart\Java\gradle\bin

#2. 在每个项目的"build.gradle"文件里设置:
repositories {
    maven{ url "http://SVN:8081/nexus/content/groups/public"}
    mavenCentral()
    mavenLocal()
}
#3. 降低daemon的空闲时间
修改%GRADLE_USER_HOME%\gradle.properties
添加上-Dorg.gradle.daemon.idletimeout=300000

缺省值在gradle-launcher-2.2.1.jar文件里的org.gradle.launcher.daemon.configuration.DaemonParameters.class
注意: org.gradle.launcher.daemon.configuration.GradleProperties.java文件里有gradle.properties的key名称!

#4. 停止所有的daemon:
cd /d "%~dp0"
cd /d %GRADLE_USER_HOME%\wrapper\dists\gradle-2.2.1-bin\88n1whbyjvxg3s40jzz5ur27\gradle-2.2.1\bin
.\gradle.bat --stop
